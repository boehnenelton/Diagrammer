    // ── HIERARCHY UTILS ──────────────────────────────────
    function genOf(id) {
        let g = 0, cur = S.shapes.find(s => s.id === id);
        while (cur && cur.parentId) {
            cur = S.shapes.find(s => s.id === cur.parentId);
            if (++g > 200) break;
        }
        return g;
    }
    function childrenOf(pid) { return S.shapes.filter(s => s.parentId === pid); }
    function descendants(pid, out = new Set()) {
        childrenOf(pid).forEach(c => { out.add(c.id); descendants(c.id, out); });
        return out;
    }
    function hiddenSet() {
        const h = new Set();
        S.shapes.forEach(s => { if (s.collapsed) descendants(s.id).forEach(id => h.add(id)); });
        return h;
    }

    // ── SNAP ─────────────────────────────────────────────
    function snp(v) { return S.snap ? Math.round(v / S.snapSz) * S.snapSz : v; }

    // Tier-based default sizes
    const TIER_SIZE = [
        {w:200,h:120}, {w:160,h:100}, {w:130,h:80}, {w:110,h:70}, {w:90,h:60}
    ];
    function tsz(gen) { return TIER_SIZE[Math.min(gen, TIER_SIZE.length-1)]; }

    function addShape(cx, cy) {
        if (cx === undefined) {
            const r = D.vp.getBoundingClientRect();
            cx = (D.vp.scrollLeft + r.width/2) / S.zoom - 100;
            cy = (D.vp.scrollTop  + r.height/2) / S.zoom - 60;
        }
        const sz = tsz(0);
        const s = {
            id:'s'+(S.nid++), x:snp(cx), y:snp(cy), w:sz.w, h:sz.h,
            color:'#334', fontColor:'auto',

        };

        sel([s.id]); render(); renderObj();
    }

    function addChildNode() {
        if (S.selIds.length !== 1) return;
        const pid = S.selIds[0];
        const par = S.shapes.find(s => s.id === pid);
        if (!par) return;
        const gen = genOf(pid) + 1;
        const sz  = tsz(gen);
        const pos = autoPlace(par, sz);

        const CCOLS = ['#2a3a5c','#1a4a2a','#4a3a1a','#3a1a4a','#334'];
        const color = CCOLS[Math.min(gen-1, CCOLS.length-1)];

        const s = {
            id:'s'+(S.nid++), x:pos.x, y:pos.y, w:sz.w, h:sz.h,
            color, fontColor:'auto',
            label:'Node '+S.nid, text:'',
            parentId:pid, collapsed:false
        };
        S.shapes.push(s);

        // Auto-connect parent bottom-center → child top-center
        S.connectors.push({
            id:'c'+Date.now()+Math.random().toString(36).slice(2),
            from:{shapeId:pid, anchorIndex:5},
            to:{shapeId:s.id,  anchorIndex:1},
            flow:'forward'
        });

        sel([s.id]); render(); renderObj();
    }

    function autoPlace(par, sz) {
        const GAP_Y=80, GAP_X=20;
        const sibs = childrenOf(par.id);
        let x, y;
        if (!sibs.length) {
            x = par.x + par.w/2 - sz.w/2;
            y = par.y + par.h + GAP_Y;
        } else {
            const re = sibs.reduce((m,s)=>Math.max(m,s.x+s.w), 0);
            x = re + GAP_X; y = sibs[0].y;
        }
        x = snp(x); y = snp(y);
        // Collision nudge
        for (let i=0; i<25; i++) {
            const hit = S.shapes.find(s =>
                s.x < x+sz.w+8 && s.x+s.w > x-8 &&
                s.y < y+sz.h+8 && s.y+s.h > y-8
            );
            if (!hit) break;
            hit.x = snp(hit.x + sz.w + GAP_X);
        }
        return {x, y};
    }

    function makeRoot() {
        S.selIds.forEach(id => {
            const s = S.shapes.find(x=>x.id===id);
            if (s) s.parentId = null;

        render(); renderObj(); updateUI();
    }

    }

    // ── SELECTION ────────────────────────────────────────
    function sel(ids) {
        S.selConn = null;
        if (S.multi) {
            ids.forEach(id => {
                if (S.selIds.includes(id)) S.selIds = S.selIds.filter(x=>x!==id);
                else S.selIds.push(id);
            });
        } else {
            S.selIds = ids;
        }
        updateUI(); render();
    }

    function selFromPanel(e, id) {
        e.stopPropagation();
        sel([id]);
        const s = S.shapes.find(x=>x.id===id);
        if (s) {
            D.vp.scrollLeft = Math.max(0, s.x * S.zoom - 120);
            D.vp.scrollTop  = Math.max(0, s.y * S.zoom - 120);
        }
        renderObj();
    }

    // ── DRAG (THE FIX) ───────────────────────────────────
    // Key: e.preventDefault() on pointerdown stops the browser claiming
    // the touch gesture as a viewport scroll. No pointer capture is used —
    // it gets invalidated when render() destroys the SVG element mid-drag.
    // Window-level pointermove/pointerup are sufficient.

    function onShapeDown(e, id) {
        e.stopPropagation();
        e.preventDefault();   // ← THE FIX: prevents viewport scroll claiming the gesture

        if (!S.multi && !S.selIds.includes(id)) {
            S.selIds = [id]; updateUI(); render();
        }

        const movIds = [...S.selIds];
        S.drag = {
            startX: e.clientX, startY: e.clientY, active: false,
            init: movIds.map(mid => {
                const s = S.shapes.find(x=>x.id===mid);
                return { id:mid, x:s.x, y:s.y };
            })
        };
    }

    function onMove(e) {
        if (!S.drag) return;
        const dx = (e.clientX - S.drag.startX) / S.zoom;
        const dy = (e.clientY - S.drag.startY) / S.zoom;
        if (!S.drag.active && Math.hypot(dx,dy) > 3) S.drag.active = true;
        if (S.drag.active) {
            e.preventDefault();
            S.drag.init.forEach(p => {
                const s = S.shapes.find(x=>x.id===p.id);
                if (s) { s.x = snp(p.x+dx); s.y = snp(p.y+dy); }
            });
            render();
        }
    }

    function onUp() {
        S.drag = null;
        renderObj();
    }

    function onBg(e) {
        if (e.target === D.svg) {
            S.selIds = []; S.selConn = null; S.pending = null;
            updateUI(); render();
        }
    }

    // ── ANCHORS & CONNECTORS ─────────────────────────────
    function onAnchorDown(e, shapeId, idx) {
        e.stopPropagation(); e.preventDefault();
        if (S.pending) {
            if (S.pending.shapeId !== shapeId) {
                const p = S.pending;
                const dup = S.connectors.some(c =>
                    (c.from.shapeId===p.shapeId && c.from.anchorIndex===p.anchorIndex &&
                     c.to.shapeId===shapeId && c.to.anchorIndex===idx) ||
                    (c.from.shapeId===shapeId && c.from.anchorIndex===idx &&
                     c.to.shapeId===p.shapeId && c.to.anchorIndex===p.anchorIndex)
                );
                if (!dup) {
                    S.connectors.push({
                        id:'c'+Date.now()+Math.random().toString(36).slice(2),
                        from:p, to:{shapeId, anchorIndex:idx}, flow:'none'
                    });
                } else {
                    D.stl.innerText = '⚠ Duplicate blocked';
                    setTimeout(()=>{ D.stl.innerText='Ready'; }, 2000);
                }
            }
            S.pending = null; D.stl.innerText='Ready';
        } else {
            S.pending = {shapeId, anchorIndex:idx};
            D.stl.innerText = 'Tap target anchor…';
        }
        render();
    }

    function onConnDown(e, id) {
        e.stopPropagation();
        S.selIds = []; S.selConn = id;
        updateUI(); render();
    }

    // ── RENDER ───────────────────────────────────────────
    const GEN_COL = ['transparent','#2a6496','#2e7d32','#e65100','#6a1a8a'];

    function render() {
        const hidden = hiddenSet();
        let h = `<defs>
            <marker id="me"  markerWidth="10" markerHeight="10" refX="9" refY="3" orient="auto"><path d="M0,0 L0,6 L9,3 z" fill="#888"/></marker>
            <marker id="ms"  markerWidth="10" markerHeight="10" refX="1" refY="3" orient="auto"><path d="M9,0 L9,6 L0,3 z" fill="#888"/></marker>
            <marker id="mes" markerWidth="10" markerHeight="10" refX="9" refY="3" orient="auto"><path d="M0,0 L0,6 L9,3 z" fill="#DE2626"/></marker>
            <marker id="mss" markerWidth="10" markerHeight="10" refX="1" refY="3" orient="auto"><path d="M9,0 L9,6 L0,3 z" fill="#DE2626"/></marker>
        </defs>`;

        // Connectors
        S.connectors.forEach(c => {
            const s1 = S.shapes.find(x=>x.id===c.from.shapeId);
            const s2 = S.shapes.find(x=>x.id===c.to.shapeId);
            if (!s1||!s2||hidden.has(s1.id)||hidden.has(s2.id)) return;
            const p1 = anchorPos(s1, c.from.anchorIndex);
            const p2 = anchorPos(s2, c.to.anchorIndex);
            const sel = S.selConn === c.id;
            let mk = '';
            if (c.flow==='forward'||c.flow==='bidirectional') mk+=` marker-end="${sel?'url(#mes)':'url(#me)'}"`;
            if (c.flow==='backward'||c.flow==='bidirectional') mk+=` marker-start="${sel?'url(#mss)':'url(#ms)'}"`;
            h += `<line x1="${p1.x}" y1="${p1.y}" x2="${p2.x}" y2="${p2.y}" class="conn${sel?' sel':''}" ${mk} pointer-events="none"/>`;
            h += `<line x1="${p1.x}" y1="${p1.y}" x2="${p2.x}" y2="${p2.y}" class="conn-hit" onpointerdown="app.connDown(event,'${c.id}')"/>`;
        });

        // Shapes
        S.shapes.forEach(s => {
            if (hidden.has(s.id)) return;
            const isSel = S.selIds.includes(s.id);
            const tc   = textCol(s);
            const gen  = genOf(s.id);
            const sc   = GEN_COL[Math.min(gen, GEN_COL.length-1)];
            const kids = childrenOf(s.id);

            h += `<g transform="translate(${s.x},${s.y})" class="shape-group" onpointerdown="app.shapeDown(event,'${s.id}')">`;
            h += `<rect width="${s.w}" height="${s.h}" fill="${s.color}" class="shape-rect ${isSel?(S.multi?'msel':'sel'):''}"/>`;
            if (gen>0) h += `<rect x="0" y="0" width="${s.w}" height="5" fill="${sc}" pointer-events="none"/>`;

            const tp = gen>0 ? 9 : 4;
            h += `<foreignObject width="${s.w}" height="${s.h}" style="pointer-events:none;overflow:hidden;">
                <div xmlns="http://www.w3.org/1999/xhtml" class="shape-text-wrap" style="color:${tc};padding-top:${tp}px;">
                    <div class="shape-label">${esc(s.label)}</div>
                    ${s.text?`<div class="shape-body">${esc(s.text)}</div>`:''}
                </div>
            </foreignObject>`;

            // Collapse/expand button
            if (kids.length) {
                const BW=24,BH=20,BX=s.w-BW-3,BY=3;
                const ico = s.collapsed ? '▶' : '▼';
                const bf  = s.collapsed ? '#DE2626' : 'rgba(0,0,0,.45)';
                h += `<g onpointerdown="app.toggleCollapse(event,'${s.id}')" style="cursor:pointer;">
                    <rect x="${BX}" y="${BY}" width="${BW}" height="${BH}" rx="4" fill="${bf}" stroke="#555" stroke-width="0.5"/>
                    <text x="${BX+BW/2}" y="${BY+14}" text-anchor="middle" fill="white" font-size="10" font-family="monospace" pointer-events="none">${ico}</text>
                </g>`;
            }

            // Anchors (show when selected or pending link)
            if (isSel || S.pending) {
                for (let i=0; i<8; i++) {
                    const p = relAnch(s.w, s.h, i);
                    const ac = (S.pending?.shapeId===s.id && S.pending.anchorIndex===i) ? ' active' : '';
                    h += `<circle cx="${p.x}" cy="${p.y}" class="anc-vis${ac}"/>`;
                    h += `<circle cx="${p.x}" cy="${p.y}" class="anc-hit" onpointerdown="app.anchorDown(event,'${s.id}',${i})"/>`;
                }
            }
            h += `</g>`;
        });

        D.svg.innerHTML = h;
    }

    function renderObj() {
        if (!D.ol || !S.objOpen) return;
        const hidden = hiddenSet();
        let h = '', n = 0;

        function row(s, depth) {
            n++;
            const isSel = S.selIds.includes(s.id);
            const isH   = hidden.has(s.id);
            const kids  = childrenOf(s.id);
            const tb    = depth>0 ? `<span class="tier-badge">T${depth}</span>` : '';
            const ci    = kids.length ? (s.collapsed?'▶ ':'▼ ') : '';
            const kc    = kids.length ? `<span class="obj-cnt">(${kids.length})</span>` : '';
            h += `<div class="obj-item${isSel?' active':''}${isH?' hidden-node':''}"
                style="padding-left:${6+depth*13}px;"
                onpointerdown="app.selPanel(event,'${s.id}')">
                ${tb}<span class="obj-lbl">${ci}${esc(s.label)}</span>${kc}
            </div>`;
            kids.forEach(c => row(c, depth+1));
        }

        const valid = new Set(S.shapes.map(s=>s.id));
        S.shapes.filter(s=>!s.parentId||!valid.has(s.parentId)).forEach(s=>row(s,0));
        D.ol.innerHTML = h || `<div style="color:#888;padding:10px;text-align:center;font-size:12px;">No nodes</div>`;
        D.oc.textContent = `${n} node${n!==1?'s':''}`;
    }

    function setupObjDrag() {
        const hdr = document.getElementById('obj-hdr');
        let dg = null;
        hdr.addEventListener('pointerdown', e => {
            const r = D.op.getBoundingClientRect();
            D.op.style.bottom = 'auto';
            D.op.style.top  = r.top+'px';
            D.op.style.left = r.left+'px';
            dg = {sx:e.clientX,sy:e.clientY,ox:r.left,oy:r.top};
            try { hdr.setPointerCapture(e.pointerId); } catch(_){}
            e.preventDefault();
        });
        hdr.addEventListener('pointermove', e => {
            if (!dg) return;
            const W=D.op.offsetWidth, H=D.op.offsetHeight;
            D.op.style.left = Math.max(0,Math.min(innerWidth-W,  dg.ox+e.clientX-dg.sx))+'px';
            D.op.style.top  = Math.max(0,Math.min(innerHeight-H, dg.oy+e.clientY-dg.sy))+'px';
        });
        hdr.addEventListener('pointerup',     ()=>{ dg=null; });
        hdr.addEventListener('pointercancel', ()=>{ dg=null; });
    }

        if (S.selConn) {
            S.connectors = S.connectors.filter(c=>c.id!==S.selConn);

    // ── UTILS ────────────────────────────────────────────
    function anchorPos(s,i) {
        const xm=s.x+s.w/2, ym=s.y+s.h/2;
        return [{x:s.x,y:s.y},{x:xm,y:s.y},{x:s.x+s.w,y:s.y},{x:s.x+s.w,y:ym},
                {x:s.x+s.w,y:s.y+s.h},{x:xm,y:s.y+s.h},{x:s.x,y:s.y+s.h},{x:s.x,y:ym}][i] || {x:xm,y:ym};
    }
    function relAnch(w,h,i) {
        const xm=w/2,ym=h/2;
        return [{x:0,y:0},{x:xm,y:0},{x:w,y:0},{x:w,y:ym},
                {x:w,y:h},{x:xm,y:h},{x:0,y:h},{x:0,y:ym}][i] || {x:xm,y:ym};
    }
    function textCol(s) {
        if (s.fontColor!=='auto') return s.fontColor;
        const hx=s.color.replace('#','');
        const r=parseInt(hx.substr(0,2),16),g=parseInt(hx.substr(2,2),16),b=parseInt(hx.substr(4,2),16);
        return ((r*299+g*587+b*114)/1000)>=128 ? '#000' : '#fff';
    }
    function esc(str) {
        return String(str||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
    }

            const l=document.createElement('div'); l.className='tick tick-v'; l.style.top=(i*S.zoom)+'px'; l.innerText=i; D.rl.appendChild(l);
        }
    }

    }

    // ── BEJSON IO ────────────────────────────────────────
    function toBEJSON() {
        return {
            Format:"BEJSON", Format_Version:"104db", Format_Creator:"Elton Boehnen",
            Records_Type:["Shape","Connector"],
            Fields:[
                {name:"Record_Type_Parent",type:"string"},
                {name:"id",type:"string",Record_Type_Parent:"Shape"},
                {name:"x",type:"number",Record_Type_Parent:"Shape"},
                {name:"y",type:"number",Record_Type_Parent:"Shape"},
                {name:"w",type:"number",Record_Type_Parent:"Shape"},
                {name:"h",type:"number",Record_Type_Parent:"Shape"},
                {name:"label",type:"string",Record_Type_Parent:"Shape"},
                {name:"color",type:"string",Record_Type_Parent:"Shape"},
                {name:"fontColor",type:"string",Record_Type_Parent:"Shape"},
                {name:"text",type:"string",Record_Type_Parent:"Shape"},
                {name:"parentId",type:"string",Record_Type_Parent:"Shape"},
                {name:"generation",type:"integer",Record_Type_Parent:"Shape"},
                {name:"collapsed",type:"boolean",Record_Type_Parent:"Shape"},
                {name:"id",type:"string",Record_Type_Parent:"Connector"},
                {name:"from",type:"string",Record_Type_Parent:"Connector"},
                {name:"fromIdx",type:"integer",Record_Type_Parent:"Connector"},
                {name:"to",type:"string",Record_Type_Parent:"Connector"},
                {name:"toIdx",type:"integer",Record_Type_Parent:"Connector"},
                {name:"flow",type:"string",Record_Type_Parent:"Connector"}
            ],
            Values:[
                ...S.shapes.map(s=>[
                    "Shape",s.id,s.x,s.y,s.w,s.h,s.label,s.color,s.fontColor,s.text,
                    s.parentId||null,genOf(s.id),s.collapsed||false,
                    null,null,null,null,null,null
                ]),
                ...S.connectors.map(c=>[
                    "Connector",
                    null,null,null,null,null,null,null,null,null,null,null,null,
                    c.id,c.from.shapeId,c.from.anchorIndex,c.to.shapeId,c.to.anchorIndex,c.flow
                ])
            ]
        };
    }

    function fromBEJSON(j, nameHint) {
        S.shapes=[]; S.connectors=[];
        if (nameHint) S.name = nameHint.replace(/\.json$/i, '');
        else if (j.Diagram_Name) S.name = j.Diagram_Name;
        const m={};
        (j.Fields||[]).forEach((f,i)=>m[f.name+(f.Record_Type_Parent?'_'+f.Record_Type_Parent:'')]=i);
        (j.Values||[]).forEach(v=>{
            if (v[0]==="Shape") {
                S.shapes.push({
                    id:v[m["id_Shape"]], x:v[m["x_Shape"]]||0, y:v[m["y_Shape"]]||0,
                    w:v[m["w_Shape"]]||100, h:v[m["h_Shape"]]||100,
                    label:v[m["label_Shape"]]||'', color:v[m["color_Shape"]]||'#333',
                    fontColor:v[m["fontColor_Shape"]]||'auto', text:v[m["text_Shape"]]||'',
                    parentId:v[m["parentId_Shape"]]||null, collapsed:v[m["collapsed_Shape"]]||false
                });
            } else if (v[0]==="Connector") {
                S.connectors.push({
                    id:v[m["id_Connector"]],
                    from:{shapeId:v[m["from_Connector"]],anchorIndex:v[m["fromIdx_Connector"]]},
                    to:{shapeId:v[m["to_Connector"]],anchorIndex:v[m["toIdx_Connector"]]},
                    flow:v[m["flow_Connector"]]||'none'
                });
            }
            // Group rows from v25 silently skipped
        });
        S.nid = S.shapes.reduce((mx,s)=>Math.max(mx,(parseInt(s.id.replace(/\D/g,''))||0)+1),S.nid);
        updateUI(); render(); renderObj();
    }

    function exportData() {
        const b=new Blob([JSON.stringify(toBEJSON(),null,2)],{type:"application/json"});
        const a=document.createElement('a');
        a.href=URL.createObjectURL(b);
        a.download=S.name.replace(/[^a-z0-9]/gi,'_').toLowerCase()+'.json';
        a.click();
    }
    function importData(inp) {
        const f=inp.files[0]; if(!f) return;
        const r=new FileReader();
        r.onload=e=>{ try{fromBEJSON(JSON.parse(e.target.result), f.name);}catch(e){alert('Import error: '+e.message);} };
        r.readAsText(f);
    }

    async function generateDiagram() {
        const key = document.getElementById('ai-key').value;
        const model = document.getElementById('ai-model').value;
        const prompt = document.getElementById('ai-prompt').value;
        const status = document.getElementById('ai-status');
        const mode = document.querySelector('input[name="ai-mode"]:checked').value;

        if (!key) { alert("Please enter an API Key"); return; }
        if (!prompt) { alert("Please enter a prompt"); return; }

        status.innerText = "Generating...";
        status.style.color = "var(--accent)";

        const systemInstruction = `You are the BEJSON Diagrammer AI. Your core mission is to generate complex node diagrams following the strict BEJSON 104db tabular standard.

### DATA STRUCTURES (STRICT):
1. Shapes (Entity: 'Shape'):
   - id: unique string starting with 's' (e.g. s1, s2)
   - x, y: numbers (MUST be multiples of 50)
   - w, h: numbers (width/height, e.g. 200, 120)
   - color, fontColor: hex strings
   - label: Title string
   - text: Body string (use \\n for line breaks)
   - parentId: id of parent shape or null
   - collapsed: boolean

2. Connectors (Entity: 'Connector'):
   - id: unique string starting with 'c' (e.g. c1, c2)
   - from, to: { shapeId: string, anchorIndex: number }
   - anchorIndex: 0=Top, 1=Right, 2=Bottom, 3=Left.
   - flow: 'none', 'forward', 'backward', 'bidirectional'

### RULES:
- Respond ONLY with a clean JSON object: { "shapes": [...], "connectors": [...] }.
- ${mode === 'replace' ? 'Mode: OVERWRITE. Create a completely new diagram.' : 'Mode: APPEND. Add to existing with unique IDs.'}
- Spacing: Ensure nodes are at least 200px apart.`;

        try {
            const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${key}`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: prompt }] }],
                    system_instruction: { parts: [{ text: systemInstruction }] }
                })
            });

            if (!res.ok) throw new Error("API Error: " + res.status);
            const data = await res.json();
            const rawText = data.candidates[0].content.parts[0].text;
            const match = rawText.match(/\{[\s\S]*\}/);
            if (!match) throw new Error("No JSON in response");
            
            const diagram = JSON.parse(match[0]);
            if (mode === 'replace') { S.shapes = []; S.connectors = []; }

            const toArr = (x) => Array.isArray(x) ? x : (x && typeof x === 'object' ? Object.values(x) : []);
            const shapes = toArr(diagram.shapes);
            const connectors = toArr(diagram.connectors);

            shapes.forEach(s => { if (s && s.id && !S.shapes.some(x => x.id === s.id)) S.shapes.push(s); });
            connectors.forEach(c => { if (c && c.id && !S.connectors.some(x => x.id === c.id)) S.connectors.push(c); });

            status.innerText = "Success!";
            status.style.color = "#4CAF50";
            setTimeout(() => { status.innerText = "AI Ready"; status.style.color = "#666"; }, 3000);
            render(); renderObj(); updateUI();
        } catch (e) {
            status.innerText = "Error: " + e.message;
            status.style.color = "#F44336";
        }
    }
